<script>
  function toggleLike(btn) {
    const countSpan = btn.querySelector('span');
    const liked = btn.classList.toggle('liked');
    countSpan.textContent = liked
      ? parseInt(countSpan.textContent) + 1
      : parseInt(countSpan.textContent) - 1;
  }

  function checkAnswer() {
    const input = document.getElementById('puzzle-input').value.trim().toLowerCase();
    const feedback = document.getElementById('answer-feedback');

    if (input === 'halpert') {
      feedback.className = 'answer-feedback correct';
      feedback.innerHTML = '<span class="confetti-emoji">🎉</span> Correct! The restaurant is <strong>Halpert\'s</strong>! Well done!';
    } else if (input === '') {
      feedback.className = 'answer-feedback incorrect';
      feedback.textContent = 'Please enter an answer first.';
    } else {
      feedback.className = 'answer-feedback incorrect';
      feedback.textContent = '✗ Not quite — keep digging through those reviews!';
    }
  }

  document.getElementById('puzzle-input').addEventListener('keydown', function(e) {
    if (e.key === 'Enter') checkAnswer();
  });
</script>
